def matxMultiply(A, B):
    n = len(A)
    m = len(A[0])
    m1 = len(B)
    p = len(B[0])
    if m != m1:
        raise ValueError
    else:
        C = [[] for q in range(n)]
        for i in range(n):
            for k in range(p):
                element = 0
                for j in range(m):
                    element += A[i][j] * B[j][k]
                C[i].append(element)
        return C

    n = len(A)
    m = len(A[0])
    k = len(b[0])
    Ab = []
    for i in range(n):
        list = []
        for j in range(m):

""" Gaussian Jordan  Ax = b.
    
        A:  
        b: 
        decPts: 4
        epsilon: 0 1.0e-16
        
     x  Ax = b 
    None Ab 
    None A 
"""
def gj_Solve(A, b, decPts=4, epsilon=1.0e-16):
    if len(A) == len(b):
        Ab = augmentMatrix(A, b)
        n = len(Ab)
        m = len(Ab[0])
        for j in range(m - 1):
            j_list = []
            for i in range(j,n):
                j_list.append(abs(Ab[i][j]))
            if max(j_list) < epsilon:
                return None
            else:
                p = j_list.index(max(j_list)) + j
                swapRows(Ab, j, p)
                main_element = Ab[j][j]
                scale = 1/main_element
                scaleRow(Ab, j, scale)
                for k in range(n):
                    if k == j:
                        continue
                    else:
                        scale1 = -(Ab[k][j])
                        addScaledRow(Ab, k, j, scale1)
    else:
        return None
    matxRound(Ab, decPts)
    x = [[ix] for ix in transpose(Ab)[-1]]
    return x
